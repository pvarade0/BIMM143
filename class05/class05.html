 x <- rnorm(1000)

#lwd = line thickness

#ylim = setting the y axis from 2 to 10
plot(baby, type = "o", pch=15, cex= 1.5, lwd=2, ylim= c(2,10))

#xlab/ylab/main = naming x axis/y axis/whole plot
